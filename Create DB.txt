CREATE TABLE [dbo].[EmailRecipient] (

    [Address] NVARCHAR (MAX) NULL,
    CONSTRAINT [PK_EmailRecipient_Id] PRIMARY KEY CLUSTERED ([Id] ASC)
);

    [Id]    INT            IDENTITY (1, 1) NOT NULL,
    [Email] NVARCHAR (MAX) NOT NULL,

    CONSTRAINT [PK_EmailSender_Id] PRIMARY KEY CLUSTERED ([Id] ASC)
);
